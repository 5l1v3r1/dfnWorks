	    else:
		raise AttributeError(str(key) + ' is not a valid attribute for ' + self.__class__.__name__)

	def _freeze(self):
            """
            Prevents adding new attributes to a class.
            """
            self.__frozen = True

	def _unfreeze(self):
	    """
            Allows adding new attributes to classes.
            """
            self.__frozen = False
